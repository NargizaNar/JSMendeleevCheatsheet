## String + String 
## Example 1 
const firstName = "Lola";
const city = "Copa-Copacabana";
const job = "showgirl";
console.log(`My name is ${firstName}. I live in ${city} and I am a ${job}`);

## Example 2 
const strPrefix = "Py";

const str = "thon";

if (str.substring(0, 2) == "Py") {
    console.log(str);
} else {
    console.log(`${strPrefix}${str}`);
}
